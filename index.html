import java.util.Scanner; public class JapaneseStudyApp { public static void main(String[] args) { Scanner scanner = new Scanner(System.in); // 1. 단어 데이터 (한국어 뜻, 일본어 정답 순서) // [사진 1: 기초 명사 및 대명사] String[][] unit1 = { {"개", "犬(いぬ)"}, {"발, 다리", "足(あし)"}, {"손", "手(て)"}, {"생선", "魚(さかな)"}, {"고양이", "猫(ねこ)"}, {"얼굴", "顔(かお)"}, {"책상", "机(つくえ)"}, {"나, 저 (남자가 가볍게)", "おれ"}, {"벚꽃", "桜(さくら)"} }; // [사진 2: 숫자와 인사말] String[][] unit2 = { {"0", "ゼロ / れい"}, {"1", "いち"}, {"2", "に"}, {"3", "さん"}, {"안녕하세요(아침)", "おはよう"}, {"안녕하세요(점심)", "こんにちは"}, {"감사합니다", "ありがとうございます"}, {"미안합니다", "すみません"} }; // [사진 3: 자기소개 및 취미] String[][] unit3 = { {"취미", "趣味(しゅみ)"}, {"요리", "料理(りょうり)"}, {"독서", "読書(どくしょ)"}, {"처음 뵙겠습니다", "はじめまして"}, {"~라고 합니다", "~ともうします"}, {"진실은 언제나 하나!", "真実はいつもひとつ！"} }; // 전체 단어장을 배열에 담기 String[][][] allUnits = {unit1, unit2, unit3}; System.out.println("=== 2026 일본어 단어 암기 프로그램 ==="); System.out.println("책에 답을 적은 후 [Enter]를 눌러 정답을 확인하세요!"); for (int i = 0; i < allUnits.length; i++) { System.out.println("\n--- [단어장 " + (i + 1) + "단계 시작] ---"); for (String[] word : allUnits[i]) { System.out.print("[문제] " + word[0] + " -> (책에 쓴 뒤 Enter)"); scanner.nextLine(); // 사용자가 Enter 누를 때까지 대기 System.out.println(">> 정답: " + word[1]); System.out.println("-----------------------------------"); } if (i < allUnits.length - 1) { System.out.println("\n" + (i + 1) + "단계를 마쳤습니다. 다음 단계로 가려면 Enter!"); scanner.nextLine(); } } System.out.println("\n모든 단어 학습이 끝났습니다! 수고하셨습니다."); scanner.close(); } }
